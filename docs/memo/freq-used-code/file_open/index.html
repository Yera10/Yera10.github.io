      내장함수 open으로 파일 읽기 (txt, csv, json) #

      readline( ) 함수 사용 #

      제너레이터처럼 한줄씩 꺼내서 str 타입으로 읽어들인다.
      (txt, csv, json 파일도 가능)

      f = open("d:\\data\\emp3.csv", "r")

file = []
while True:
	line = f.readline()    # readline: 한줄씩 읽어들임
	if not line: break
	file.append(line)    # readline: 한줄을 문자열로 반환

f.close()
print(file)    # file : [ str, str, ..., str ] 의 형태

      readlines( ) 함수 사용 #

      readline( ) 함수와는 다르게 list 타입으로 반환한다.
      반복문을 사용하지 않아도 된다.

      f = open('d:\\data\\emp3.csv', 'r')
file = f.readlines()
f.close()     # [ str, str, ..., str ]

      read( ) 함수 사용 #

      파일 전체를 str 타입으로 반환

      f = open('d:\\data\\emp3.csv', 'r')
file = f.read()    # 전체 str 형태로 반환
f.close()

      with문 사용하기 #

      with open('d:\\data\\emp3.csv', 'r') as f:
	file = f.read()    # str

      with open('d:\\data\\emp3.csv', 'r') as f:
	file = f.readlines()    # file : [ str, str, ..., str ]

      이미지 파일 #

      PIL 라이브러리 사용해서 읽기 #

      from PIL import Image
img = Image.open('파일경로')
img.show()

      PIL.Image, numpy, matplotlib을 사용해서 읽기 #

      import PIL.Image as pilimg
import numpy as np
import matplotlib.pyplot as plt

im = pilimg.open('파일경로')
pix = np.array(im)
plt.imshow(pix)

      from PIL import Image
img = Image.open('파일경로')
img.save('파일경로')

      pyarrow 라이브러리 사용 #

      파일 용량이 커서 pandas나 csv로 열 수 없을 때, pyarrow를 사용하면 좋다.
      속도도 더 빠르다.

      import pandas as pd
from pyarrow import csv

tb = csv.read_csv('csv파일 경로')
df = tb.to_pandas()
